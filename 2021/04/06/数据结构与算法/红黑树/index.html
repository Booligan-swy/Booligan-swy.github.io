红黑树

定义：

红黑树是基于2-3树思想的内含有红黑链的二叉查找树，并满足以下三种条件：

  • 1. 红色链接必须是左链接；
  • 2. 任何一个结点不可能有两个红色链接；
  • 3. 该树是完美黑色匹配，也即任意的空连到根结点的路径上的黑链接数目是相同的。

红黑树的一些基本操作

1. 左旋

当某个结点的左子结点的链接是黑色的，而其右子结点是红色的（不满足定义中的第一个条件）则需要对其进行左旋操作。

左旋的过程（前提条件当前结点为h，而其右子结点为x）：

  • 1. 让x的左子结点成为h的右子结点；
  • 2. 让h成为x的左子结点；
  • 3. 让h的color属性成为x的color属性；
  • 4. 让h的color属性置为红色。

2.右旋

当某个结点的左子结点的链接是红色的，而其左子结点的左子结点也是红色的（不满足定义中的第二个条件）则需要对其进行右旋操作。

右旋的过程（前提条件当前结点为h，而其左子结点为x）：

  • 1. 让x的右子结点成为h的左子结点；
  • 2. 让h成为x的右子结点；
  • 3. 让h的color属性成为x的color属性；
  • 4. 让h的color属性置为红色。

3.颜色反转

当某个结点的左子结点的链接是红色的，而其右子结点也是红色的，则需要将其两个子结点的链接设置为黑色，而其父链接设置为红色。

代码块：

/*
红黑树的CPP实现
*/
#include<iostream>
#include<string>

using namespace std;

template<class Key, class Value>
class Node {
public:
    Key key;
    Value value;

    //记录左子结点和右子结点
    Node* left;
    Node* right;

    bool color;

    Node(Key key, Value value, Node* left, Node* right, bool color) {
        this->key = key;
        this->value = value;
        this->left = left;
        this->right = right;
        this->color = color;
    }
};

template<class Key, class Value>
class RedBlackTree {
public:
    Node<Key,Value>* root;
    int N;

    //红黑树标识： 红 RED true， 黑 BLACK false
    const bool RED = true;
    const bool BLACK = false;

public:
    RedBlackTree(Key key, Value value, Node<Key, Value>* left, Node<Key, Value>* right, bool color) {
        this->root = new Node<Key, Value>(key,value,left,right,color);
        this->N = 0;
    }
    
    ~RedBlackTree() {
        delete root;
    }


    //获取树中元素的个数
    int getSize() {
        return N;
    }

    //判断当前结点的父指向链接是否为红色
    bool isRed(Node<Key, Value>* x) {
        if (x == NULL) {
            return false;
        }

        return x->color == RED;
    }

    //左旋操作
    Node<Key, Value>* rotateLeft(Node<Key, Value>* h) {
        //获取h结点的右子结点x
        Node<Key, Value>* x = h->right;
        //让x结点的左子结点成为h结点的右子结点
        h->right = x->left;
        //让h结点成为x的左子结点
        x->left = h;
        //让x结点的color属性等于h结点的color属性
        x->color = h->color;
        //让h结点的color成为红色
        h->color = RED;

        return x;
    }

    //右旋操作
    Node<Key,Value>* rotateRight(Node<Key, Value>* h) {
        //获取h结点的左子结点x
        Node<Key, Value>* x = h->left;
        //让x的右子结点成为h的左子结点
        h->left = x->right;
        //让h成为x的右子结点
        x->right = h;
        //让x结点的color属性等于h结点的color属性
        x->color = h->color;
        //让h结点的color成为红色
        h->color = RED;

        return x;
    }

    //颜色变换
    void flipColors(Node<Key, Value>* h) {
        h->color = RED;

        h->left->color = BLACK;
        h->right->color = BLACK;
    }

    int compareTo(Key key1, Key key2) {
        
        return key1 - key2;
        
    }

    //在指定的树中完成插入操作，并返回添加元素后新的树
    Node<Key, Value>* put(Node<Key, Value>* h, Key key, Value value) {
        //若h为空，则直接返回一个红色的结点即可
        if (h == NULL) {
            N++;
            return new Node<Key, Value>(key, value, NULL, NULL, RED);
        }

        
        int cmp = compareTo(key, h->key);

        if (cmp < 0) {
            //继续往左
            h->left = put(h->left, key, value);
        }
        else if (cmp > 0) {
            //继续往右
            h->right = put(h->right, key, value);
        }
        else {
            h->value = value;
        }

        //进行左旋
        //当当前结点的左子结点为黑色，而右子结点为红色
        if (isRed(h->right) && !isRed(h->left)) {
            h = rotateLeft(h);
        }

        //进行右旋
        //当当前结点的左子结点为红色，而其左子结点的左子结点也为红色
        if (isRed(h->left) && isRed(h->left->left)) {
            h = rotateRight(h);
        }

        //颜色反转：当前结点的左子结点和右子结点都为红色
        if (isRed(h->left) && isRed(h->right)) {
            flipColors(h);
        }
        
        return h;
    }

    //在整棵树上完成插入操作
    void put(Key key, Value value){
        root = put(root, key, value);
        //根结点的颜色永远是黑色
        root->color = BLACK;
    }

    
    //从指定的树x中查找key对应的值
    Value get(Node<Key, Value>* x, Key key) {
        if (x == NULL) {
            //C6387
            return NULL;
        }

        //比较x结点的键和key的大小
        
        int cmp = compareTo(key, x->key);

        if (cmp > 0) {
            return get(x->right, key);
        }
        else if (cmp < 0) {
            return get(x->left, key);
        }
        else {
            return x->value;
        }
    }

    //根据key从书中找到对应的值
    Value get(Key key) {
        return get(root, key);
    }
};

int main() {
    RedBlackTree<int, string> tree(0,"",NULL,NULL,true);
    tree.put(1, "Tom");
    tree.put(2, "Tim");
    tree.put(3, "Jack");
    tree.put(4, "Smith");

    string str1 = tree.get(2);
    cout << str1 << endl;

    string str2 = tree.get(2);
    cout << str2 << endl;

    string str3 = tree.get(3);
    cout << str3<< endl;
    return 0;
}
